function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(MAIN_SHEET); // 1. 【最優先】管理画面からの各種更新（データ保存 or LINE通知） if (e.parameter.update) { const targetId = String(e.parameter.id || ''); const rId = e.parameter.id; // ───パターンA：LINE通知の送信リクエストだった場合─── if (e.parameter.notice === 'true') { sendPushNotice(rId); // LINEへの通知関数を実行 const out = ContentService.createTextOutput(JSON.stringify({ status: 'notice_sent' })); out.setMimeType(ContentService.MimeType.JSON); return out; } // ───パターンB：患者名または受付番号の保存リクエストだった場合─── const data = sheet.getDataRange().getValues(); const pName = e.parameter.patientName; let targetUserId = ''; // データベース紐付け用のユーザーIDを格納する変数 for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === targetId) { // メインシートのC列（3列目・インデックス2）からLINEのユーザーIDを取得 targetUserId = String(data[i][2]).trim(); // 患者名が送られてきている場合はメインシートに保存 if (pName !== undefined && pName !== null) { sheet.getRange(i + 1, 5).setValue(String(pName)); // E列(5列目) = 患者名 } // 受付番号が送られてきている場合はメインシートに保存 if (e.parameter.receptionNo !== undefined) { sheet.getRange(i + 1, 6).setValue(e.parameter.receptionNo); // F列(6列目) = 受付番号 } break; } } // データベースシート（DB_SHEET）への追記・更新処理（名前が送られてきている場合のみ実行） if (targetUserId && pName !== undefined && pName !== null && pName !== '') { const dbSheet = ss.getSheetByName(DB_SHEET); if (dbSheet) { const dbData = dbSheet.getDataRange().getValues(); let isUserExists = false; // データベースに同じユーザーIDがあるか確認 for (let j = 1; j < dbData.length; j++) { if (String(dbData[j][0]).trim() === targetUserId) { // ユーザーIDが見つかったら、2列目（B列）に患者名を追記・更新 dbSheet.getRange(j + 1, 2).setValue(String(pName)); isUserExists = true; break; } } // 万が一、LINE受信時にID登録が漏れていた場合のセーフティ if (!isUserExists) { dbSheet.appendRow([targetUserId, String(pName)]); } } } const out = ContentService.createTextOutput(JSON.stringify({ status: 'updated' })); out.setMimeType(ContentService.MimeType.JSON); return out; } // 2. 受付IDで行を特定してステータスを「確認済」に更新（右端の確認完了ボタン用） if (e.parameter.id) { const targetId = String(e.parameter.id); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === targetId) { sheet.getRange(i + 1, 8).setValue('確認済'); // H列 = ステータス break; } } const out = ContentService.createTextOutput(JSON.stringify({ status: 'success' })); out.setMimeType(ContentService.MimeType.JSON); return out; } // 3. 全データ取得（管理画面の定期ポーリング用） const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = []; for (let i = 1; i < data.length; i++) { const row = data[i]; const rowData = {}; headers.forEach((header, index) => { if (row[index] instanceof Date) { rowData[header] = Utilities.formatDate(row[index], 'JST', 'yyyy-MM-dd HH:mm:ss'); } else { rowData[header] = row[index]; } }); rows.push(rowData); } const out = ContentService.createTextOutput(JSON.stringify(rows)); out.setMimeType(ContentService.MimeType.JSON); return out; }
